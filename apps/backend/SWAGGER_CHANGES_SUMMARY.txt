================================================================================
DECORATOR PATTERNS APPLIED
================================================================================

Pattern 1: Protected API Endpoints
  @ApiTags('feature-tag')
  @ApiBearerAuth('JWT')
  @Controller('route')
  export class FeatureController { }

Pattern 2: API Operation with Docs
  @ApiOperation({
    summary: 'Clear action description',
    description: 'Detailed explanation'
  })
  @ApiResponse({ status: 200, description: 'Success' })
  @ApiResponse({ status: 404, description: 'Not Found' })
  async method() { }

Pattern 3: Webhook Endpoints
  @ApiTags('webhooks')
  @Post('webhook')
  @Public()
  @HttpCode(HttpStatus.OK)
  @ApiExcludeEndpoint()
  @ApiOperation({ summary: 'Internal webhook' })
  async handleWebhook() { }
